    }

    .processing { background: blue; }
    .solved { background: green; }

    .alert { color: red; font-weight: bold; }

<h2>📜 Complaint</h2>

<div class="card" id="complaintBox"></div>

<script>

// 🔐 USER ROLE (change to "user" to test)
const user = { name: "Admin", role: "admin" };

// 📦 Complaint Data
let complaint = {
  id: 1,
  name: "Ram Bahadur",
  status: "Pending",
  createdAt: new Date(new Date() - 4*24*60*60*1000), // 4 days ago
  assignedTo: "",
  history: [
    { status: "Pending", by: "System", date: new Date() }
  ]
};

// 👨‍💼 Officers
const officers = ["Ram Officer", "Shyam Officer"];

// 🔄 Status Flow
const flow = ["Pending", "Processing", "Solved"];

// 📊 Dashboard
function updateDashboard() {
  let total = 1;
  let solved = complaint.status === "Solved" ? 1 : 0;
  let percent = (solved / total) * 100;

  document.getElementById("dashboard").innerHTML = `
    Total: ${total} <br>
    Solved: ${solved} <br>
    Solved %: ${percent.toFixed(1)}%
  `;
}

// ⏰ Alert check
function checkAlert() {
  let diff = (new Date() - complaint.createdAt) / (1000*60*60*24);
  return complaint.status === "Pending" && diff >= 3;
}

// 🔄 Status change
function changeStatus() {
  if (user.role !== "admin") {
    alert("Only admin can update");
    return;
  }

  let i = flow.indexOf(complaint.status);
  let next = flow[(i + 1) % flow.length];

  complaint.status = next;

  complaint.history.push({
    status: next,
    by: user.name,
    date: new Date()
  });

  render();
}

// 👨‍💼 Assign officer
function assignOfficer(value) {
  if (user.role !== "admin") return;

  complaint.assignedTo = value;
  render();
}

// 🎨 Render UI
function render() {

  let statusClass = complaint.status.toLowerCase();

  let historyHTML = complaint.history.map(h =>
    `<div>${h.status} - ${h.by} (${new Date(h.date).toLocaleString()})</div>`
  ).join("");

  let officerHTML = user.role === "admin" ? `
    <select onchange="assignOfficer(this.value)">
      <option>Select Officer</option>
      ${officers.map(o => `<option>${o}</option>`).join("")}
    </select>
  ` : `<p>Assigned: ${complaint.assignedTo || "None"}</p>`;

  let alertHTML = checkAlert() ? `<div class="alert">⚠️ Delayed (3+ days)</div>` : "";

  document.getElementById("complaintBox").innerHTML = `
    <h3>${complaint.name}</h3>

    <div class="status ${statusClass}" onclick="changeStatus()">
      ${complaint.status}
    </div>

    ${alertHTML}

    <p><b>Assigned:</b> ${complaint.assignedTo || "None"}</p>

    ${officerHTML}

    <h4>📜 History</h4>
    ${historyHTML}
  `;

  updateDashboard();
}

// 🚀 INIT
render();

</script>
